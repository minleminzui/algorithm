/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        // dfs O(n) O(n)
        // return dfs(root, targetSum, 0);
        if (root == nullptr) {
            return false;
        }
        if (root -> left == nullptr && root -> right == nullptr) {
            return targetSum - root -> val == 0;
        }
        return hasPathSum(root -> left, targetSum - root -> val)
                || hasPathSum(root -> right, targetSum - root -> val);
    }

    bool dfs(TreeNode *root, int targetSum, int now) {
        if (root == nullptr) return false;
        now += root -> val;
        // if (now > targetSum) return false;// æœ‰è´Ÿæ•°
        if (root -> left == nullptr && root -> right == nullptr) {
            if (now == targetSum) {
                return true;
            } else {
                return false;
            }    
        }
        return dfs(root -> left, targetSum, now) || dfs(root -> right, targetSum, now);
    }
};

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
 var M = math.MinInt32
func maxPathSum(root *TreeNode) int {
    help(root)
    return M
}

func help(root *TreeNode) int {
    if root == nil {
        return 0
    }
    var left, right int
    left = max(left, help(root.Left))        
    right = max(right, help(root.Right))

    M = max(M, right + left + root.Val)

    return max(right, left) + root.Val
}

func max(a int, b int) int {
    if a > b {
        return a
    } else {
        return b
    }
}
