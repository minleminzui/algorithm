class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        // æ ˆ O(n) O(n)
        stack<int> num;
        for (auto &s : tokens) {
            if ((s[0] == '-' && s.size() > 1)|| s[0] >= '0' && s[0] <= '9') {
                num.emplace(stoi(s));
            } else {
                int a = num.top(); num.pop();
                int b = num.top(); num.pop();
                int t;
                if (s == "+") {
                    t = b + a;
                } else if (s == "-") {
                    t = b - a;
                } else if (s == "*") {
                    t = b * a;
                } else {
                    t = b / a;
                }
                num.emplace(t);
            }
        }

        return num.top();
    }
};
