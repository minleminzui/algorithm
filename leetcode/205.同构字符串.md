class Solution {
public:
    bool isIsomorphic(string s, string t) {
        // hashè¡¨ O(n) O(n)
        unordered_map<char, char> s_to_t;
        unordered_map<char, char> t_to_s;

        for (int i = 0; i < s.size(); ++i) {
            if (s_to_t.count(s[i]) || t_to_s[t[i]]) {
                if (s_to_t[s[i]] != t[i] || t_to_s[t[i]] != s[i]) {
                    return false;
                }
            } else {
                s_to_t[s[i]] = t[i];
                t_to_s[t[i]] = s[i];
            }
        }

        return true;
    }
};
