class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        // hash è¡¨
        unordered_set<char> us_col[10], us_row[10], us_mat[3][3];

        for (int i = 0; i < 9; ++i) {
            for (int j = 0; j < 9; ++j) {
                char c = board[i][j];
                if (c == '.') {
                    continue;
                }
                if (us_col[i].count(c)) {
                    return false;
                } else {
                    us_col[i].insert(c);
                }

                if (us_row[j].count(c)) {
                    return false;
                } else {
                    us_row[j].insert(c);
                }

                if (us_mat[i / 3][j / 3].count(c)) {
                    return false;
                } else {
                    us_mat[i / 3][j / 3].insert(c);
                }
            }
        }

        return true;
    }
};
