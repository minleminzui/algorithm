```cpp
        // 滑动窗口 O(n) O(1)
        int n = nums.size();
```

```go
func minSubArrayLen(target int, nums []int) int {
    sum := 0
    ans := math.MaxInt32
    for left, right := 0, 0; right < len(nums); right++ {
        sum += nums[right]
        for sum >= target {
            ans = min(right - left + 1, ans)
            sum -= nums[left]
            left++
        }
    }
    if ans == math.MaxInt32 {
        return 0
    }
    return ans
}

func min(a, b int) int {
    if a > b {
        return b
    } else {
        return a
    }
}
```
