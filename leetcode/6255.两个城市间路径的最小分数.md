class Solution {
public:
    int minScore(int n, vector<vector<int>>& roads) {
        // å›¾ dfs O(n) O(n)
        int ans = INT_MAX;
        vector<vector<pair<int, int>>> g(n + 1);
        vector<int> vis(n + 1);
        for (auto & r : roads) {
            g[r[0]].push_back({r[1], r[2]});
            g[r[1]].push_back({r[0], r[2]});
        }

        function<void(int)> dfs = [&](int o) {
            if (vis[o]) return;
            vis[o] = true;
            for (auto & p : g[o]) {
                ans = min(p.second, ans);
                dfs(p.first);
            }
        };

        dfs(1);

        return ans;

    }
};
