class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        // hashè¡¨ O(n) O(1)
        vector<int> ans;
        for (int i = 0; i < nums.size(); ++i) {
            int pos = abs(nums[i]) - 1;
            if (nums[pos] > 0) {
                nums[pos] *= -1;
            }
        }
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] > 0) {
                ans.emplace_back(i + 1);
            }
        }

        return ans;
    } 
};
