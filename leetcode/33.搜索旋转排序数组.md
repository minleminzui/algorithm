class Solution {
public:
    int search(vector<int>& nums, int target) {
        // 二分 O(logn) O(1)
        // 每次根据mid，从有序的一半寻找target，如果没找到，就将当前范围置为无序的另一半
        int n = nums.size();

        int l = 0, r = n - 1;

        while (l <= r) {
            int m = l + r >> 1;
            if (nums[m] == target) return m;
            if (nums[0] <= nums[m]) {
                if (target >= nums[0] && target < nums[m]) {
                    r = m - 1;
                } else {    
                    l = m + 1;
                }
            } else {
                if (target <= nums[n - 1] && target > nums[m]) {
                    l = m + 1;
                } else {
                    r = m - 1;
                }
            }
        }

        return  -1;
    }
};
