class Solution {
public:
    int climbStairs(int n) {
        // dp O(n) O(1)
        if (n == 1 || n == 2) return n;
        int pre = 1, now = 2;

        for (int i = 3; i <= n; ++i) {
            int t = pre + now;
            pre = now;
            now = t;
        }

        return now;
    }
};
