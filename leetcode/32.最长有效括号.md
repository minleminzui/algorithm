class Solution {
public:
    int longestValidParentheses(string s) {
        // 栈 O(n) O(n)

        stack<int> stk;
        
        int ans = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '(') {
                stk.emplace(i);
            } else {
                if (stk.size() && s[stk.top()] == '(') {
                    // ")()())"这个会错，应该这样写
                    // ans = max(ans, i - stk.top() + 1);
                    // stk.pop();
                    stk.pop();
                    ans = max(ans, i - (stk.empty() ? -1 : stk.top()));
                } else {
                    stk.emplace(i);
                }
            }
        }

        return ans;

    }
};
