class Solution {
public:
    int numSquares(int n) {
        // bfs 分析不出来
        vector<int> dp(n + 1, n);
        queue<int> q;
        dp[0] = 0, q.push(0);
        while(q.size()) {
            int t = q.front();
            q.pop();
            if(t == n) {
                break;
            }

            for(int i = 1; i * i + t <= n; ++i) {
                if(dp[i * i + t] > dp[t] + 1) {
                    dp[i * i + t] = dp[t] + 1;
                    q.push(i * i + t);
                }
            }
        }

        return dp[n];
