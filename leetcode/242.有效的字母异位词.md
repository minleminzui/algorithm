class Solution {
public:
    bool isAnagram(string s, string t) {
        // hashè¡¨ O(n) O(n) 
        if (s.size() != t.size()) return false;
        unordered_map<char, int> um;

        for (auto & c : s) {
            um[c]++;
        }

        for (auto & c : t) {
            um[c]--;
            if (um[c] < 0) {
                return false;
            }
        }

        return true;
    }
};
