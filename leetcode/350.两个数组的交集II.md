class Solution {
public:
    vector<int> intersect(vector<int>& nums1, vector<int>& nums2) {
        // hashè¡¨ O(m + n) O(min(n, m))
        unordered_map<int,int> m;
        vector<int> ans;
        for(int &i : nums1) ++m[i];
        for(int &i : nums2) {
            if(m[i]) {
                ans.emplace_back(i);
                --m[i];
            }
        }

        return ans;
    }
};
