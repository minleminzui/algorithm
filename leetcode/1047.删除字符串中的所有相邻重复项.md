class Solution {
public:
    string removeDuplicates(string s) {
        // æ ˆ O(n) O(n)
        string ans;

        for (auto c : s) {
            if (ans == "" || ans.back() != c) {
                ans.push_back(c);
            } else {
                ans.pop_back();
            }
        }

        return ans;
    }
};
