class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        // O(2 ^ n) O(n) å›æº¯
        int ans = 0;
        help(ans, 0, nums, target);
        return ans;
    }

    void help(int &ans, int dep, vector<int> &nums, int target) {
        if (dep == nums.size()) {
            if (target == 0) {
                ++ans;
            }
            return;
        }

        help(ans, dep + 1, nums, target - nums[dep]);
        help(ans, dep + 1, nums, target + nums[dep]);
    }
};
