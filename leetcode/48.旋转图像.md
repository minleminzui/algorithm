class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        // 数学 O(n ^ 2)  O(1)
        int n = matrix.size();

        // 先转置
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                swap(matrix[i][j], matrix[j][i]);
            }
        }

        // 在反转每一个行
        for (auto &v : matrix) {
            reverse(v.begin(), v.end());
        }
    }
};
