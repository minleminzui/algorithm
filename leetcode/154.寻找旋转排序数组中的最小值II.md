class Solution {
public:
    int findMin(vector<int>& nums) {
        // 二分 O(logn) O(1)
        // 首尾元素重复会导致二分的判断出错，所以pop尾部元素，直到首尾元素不一致
        while (nums.size() > 1 && nums[0] == nums.back()) {
            nums.pop_back();
        }
        int l = 0, r = nums.size() - 1;
        if (nums[l] > nums[r]) {
            while (l < r) {
                
                int m = (l + r) >> 1;
                if (nums[m] < nums[0]) {
                    r = m;
                } else {
                    l = m + 1;
                }
                
                
            }
        }

        return nums[l];
        
    }
};
