class Solution {
public:
    int findLHS(vector<int>& nums) {
        // hash è¡¨ O(n) O(n)

        unordered_map<int, int> um;

        for (int & i : nums) {
            um[i]++;
        }
        int ans = 0;
        for (auto & [val, num] : um) {
            if (um.count(val + 1)) {
                ans = max(ans, num + (um[val + 1]));
            }
        }
        return ans;
    }
};
