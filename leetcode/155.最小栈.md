class MinStack {
    stack<int> m, nums;

public:
    MinStack() {

    }
    // 单调栈，时间复杂度O(1), 空间复杂度O(n)
    void push(int val) {
        nums.push(val);
        if(m.empty() || m.top() >= val) {
            m.push(val);
        }
    }
    
    void pop() {
       if(nums.top() == m.top()) m.pop();
        nums.pop();
    }
    
    int top() {
        return nums.top();
    }
    
    int getMin() {
        return m.top();
    }
};
