class Solution {
public:
    int findMin(vector<int>& nums) {
        // 线性遍历 O(n) O(1)
        // int m = nums[0];
        // for(auto i : nums) m = min(i, m);
        // return m;

        // 二分 O(logn) O(1)
        int m, l = 0, r = nums.size() - 1;
        if(nums[r] > nums[l]) return nums[l];
        while(l < r) {
            m = (l + r) >> 1;
            if(nums[m] < nums[0]) r = m;
            else l = m + 1;
        } 

        return nums[l];
    }
};
