class Solution {
public:
    bool isValid(string s) {
        // æ ˆ O(n) O(n)
        unordered_map<char, char> m = {
            {')', '('}, {']', '['}, {'}', '{'}
        };

        stack<char> st;

        for (auto c : s) {
            if (st.size()) {
                if (st.top() == m[c]) {
                    st.pop();
                } else {
                    st.emplace(c);
                }
            } else {
                st.emplace(c);
            }
        }

        return st.empty();
    }
};
