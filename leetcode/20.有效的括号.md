```cpp
class Solution {
public:
    bool isValid(string s) {
        // æ ˆ O(n) O(n)
        unordered_map<char, char> m = {
            {')', '('}, {']', '['}, {'}', '{'}
        };

        stack<char> st;

        for (auto c : s) {
            if (st.size()) {
                if (st.top() == m[c]) {
                    st.pop();
                } else {
                    st.emplace(c);
                }
            } else {
                st.emplace(c);
            }
        }

        return st.empty();
    }
};
```

```go
func isValid(s string) bool {
    vaild := make(map[rune]rune, 0)
    vaild[')'] = '('
    vaild[']'] = '['
    vaild['}'] = '{'

    buf := []rune{}
    for _, v := range s {
        if v == ')' || v == ']' || v == '}' {
            l := len(buf)
            if l > 0 && buf[l - 1] == vaild[v] {
                buf = buf[:l - 1]
                continue
            }
        } 
        buf = append(buf, v)      
    }
    return len(buf) == 0
}
```
