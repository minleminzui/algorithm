class Solution {
public:
    void conversion(int &n) {
        int a = 0;
        while (n) {
            int num = n % 10;
            a += num * num;
            n /= 10;
        }
        n = a;
    }

    bool isHappy(int n) {
        // hash è¡¨ O(logn) O(n)
        unordered_set<int> us;
        
        while (1) {
            if (us.count(n)) return false;
            if (n == 1) break;
            us.insert(n);
            conversion(n);

        }

        return true;
    }
};
